factory WebGLMaterial.from(Material material) {

  if (material["__webglMaterial"] == null) {
    var webglMaterial = new WebGLMaterial._internal(material);
    material["__webglMaterial"] = webglMaterial;
  }
    
  return material["__webglMaterial"];
}
